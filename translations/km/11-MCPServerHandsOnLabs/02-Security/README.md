## рҹҺҜ бһўбҹ’бһңбһёбһҠбҹӮбһӣбһҳбһјбһӣбһҠбҹ’бһӢбһ¶бһ“бһ“бҹҒбҹҮбһӮбҹ’бһҡбһ”бһҠбһҺбҹ’бһҸбһ”бҹӢ

бһҳбһјбһӣбһҠбҹ’бһӢбһ¶бһ“бһ“бҹҒбҹҮбһ•бҹ’бһҸбһӣбҹӢбһҖбһ¶бһҡбһҺбҹӮбһ“бһ¶бҹҶбһҠбҹҸбһ‘бһјбһӣбҹҶбһ‘бһјбһӣбһ¶бһҷбһўбҹҶбһ–бһёбһҖбһ¶бһҡбһўбһ“бһ»бһңбһҸбҹ’бһҸбһҹбһ»бһңбһҸбҹ’бһҗбһ·бһ—бһ¶бһ–бһҗбҹ’бһ“бһ¶бһҖбҹӢбһҹбһ бһӮбҹ’бһҡбһ¶бһҹ бһ“бһ·бһ„бһ…бҹ’бһҡбһҫбһ“бһўбһҸбһ·бһҗбһ·бһҮбһ“бһҹбһҳбҹ’бһҡбһ¶бһ”бҹӢбһҳбҹүбһ¶бһҹбҹҠбһёбһ“бһҳбҹҒ MCPбҹ” бһўбҹ’бһ“бһҖбһ“бһ№бһ„бһҡбҹҖбһ“бһҡбһ…бһ“бһ¶бһ”бҹ’бһҡбһ–бҹҗбһ“бҹ’бһ’бһҠбҹӮбһӣбһҳбһ¶бһ“бһҹбһ»бһңбһҸбҹ’бһҗбһ·бһ—бһ¶бһ– бһ“бһ·бһ„бһўбһ“бһ»бһӣбҹ„бһҳбһҠбҹӮбһӣбһҖбһ¶бһҡбһ–бһ¶бһҡбһ‘бһ·бһ“бҹ’бһ“бһ“бҹҗбһҷбһӣбҹҶбһўбһ·бһҸбһ“бҹ…бһҖбҹ’бһ“бһ»бһ„бһҖбһ¶бһҡбһӣбһҖбҹӢбһҡбһ¶бһҷбһҒбһҺбҹҲбһҠбҹӮбһӣбһўбһ“бһ»бһүбҹ’бһүбһ¶бһҸбһІбҹ’бһҷбһҳбһ¶бһ“бһӣбҹҶбһ“бһ¶бҹҶбһ…бһјбһӣбһ”бһҸбҹӢбһ”бҹӮбһ“бһҖбҹ’бһ“бһ»бһ„бһ…бһ“бҹ’бһӣбҹ„бҹҮбһўбһҸбһ·бһҗбһ·бһҮбһ“бһ…бҹ’бһҡбһҫбһ“бһ“бһ¶бһҖбҹӢбҹ”

## бһҹбҹҒбһ…бһҖбҹ’бһҸбһёбһ•бҹ’бһҠбһӣбҹӢбһҳбһ»бһ“

бһҹбһ»бһңбһҸбҹ’бһҗбһ·бһ—бһ¶бһ–бһӮбһәбһҮбһ¶бһҡбһҝбһ„бһҹбҹҶбһҒбһ¶бһ“бҹӢбһ“бҹ…бһҖбҹ’бһ“бһ»бһ„бһҖбһҳбҹ’бһҳбһңбһ·бһ’бһёбһӣбһҖбҹӢбһҡбһ¶бһҷбһҠбҹӮбһӣбһҗбҹӮбһҖбҹ’бһҹбһ‘бһ·бһ“бҹ’бһ“бһ“бҹҗбһҷбһўбһҸбһ·бһҗбһ·бһҮбһ“ бһ–бҹҗбһҸбҹҢбһҳбһ¶бһ“бһ”бһ„бҹӢбһ”бҹ’бһҡбһ¶бһҖбҹӢ бһ“бһ·бһ„бһҜбһҖбһ‘бҹҒбһҹбһўбһ¶бһҮбһёбһңбһҖбһҳбҹ’бһҳбҹ” бһҳбһјбһӣбһҠбҹ’бһӢбһ¶бһ“бһ“бҹҒбҹҮбһӮбҹ’бһҡбһ”бһҠбһҺбҹ’бһҸбһ”бҹӢбһҹбҹ’бһҗбһ¶бһ”бһҸбҹ’бһҷбһҖбһҳбҹ’бһҳбһҹбһ»бһңбһҸбҹ’бһҗбһ·бһ—бһ¶бһ–бһ–бҹҒбһүбһӣбҹҒбһүбһ…бһ¶бһ”бҹӢбһ–бһёбһҖбһ¶бһҡбһ•бҹ’бһ‘бҹҖбһ„бһ•бҹ’бһ‘бһ¶бһҸбҹӢбһўбһҸбҹ’бһҸбһҹбһүбҹ’бһүбһ¶бһҺ бһ“бһ·бһ„бһўбһ“бһ»бһүбҹ’бһүбһ¶бһҸбһҖбһҳбҹ’бһҳ бһҠбһӣбҹӢбһҖбһ¶бһҡбһ”бҹҶбһ”бҹӮбһҖбһ‘бһ·бһ“бҹ’бһ“бһ“бҹҗбһҷ бһ“бһ·бһ„бһҖбһ¶бһҡбһҸбҹ’бһҡбһҪбһҸбһ–бһ·бһ“бһ·бһҸбҹ’бһҷбһҖбһ¶бһҡбһўбһ“бһ»бһӣбҹ„бһҳбҹ”

бһҷбһҫбһ„бһўбһ“бһ»бһңбһҸбҹ’бһҸбһҷбһ»бһ‘бҹ’бһ’бһҹбһ¶бһҹбҹ’бһҸбҹ’бһҡбһҖбһ¶бһҡбһ–бһ¶бһҡбһҮбһҳбҹ’бһҡбҹҮбһҮбҹ’бһҡбҹ…бһҠбҹӮбһӣбһ•бҹ’бһҹбҹҶбһӮбҹ’бһ“бһ¶бһҡбһңбһ¶бһ„бһҹбҹҒбһңбһ¶бһўбһҸбҹ’бһҸбһҹбһүбҹ’бһүбһ¶бһҺ Azure бһҹбһ»бһңбһҸбҹ’бһҗбһ·бһ—бһ¶бһ–бһҖбһҳбҹ’бһҡбһ·бһҸбһҮбһҪбһҡбһҠбһ‘бҹғ PostgreSQL бһҖбһ¶бһҡбһӮбҹ’бһҡбһ”бҹӢбһӮбҹ’бһҡбһ„бһҖбһҳбҹ’бһҡбһ·бһҸбһҖбһҳбҹ’бһҳбһңбһ·бһ’бһё бһ“бһ·бһ„бһҖбҹҶбһҺбһҸбҹӢбһ бҹҒбһҸбһ»бһҸбҹ’бһҡбһҪбһҸбһ–бһ·бһ“бһ·бһҸбҹ’бһҷбһ‘бһјбһӣбҹҶбһ‘бһјбһӣбһ¶бһҷ бһҠбһҫбһҳбҹ’бһ”бһёбһ”бһ„бҹ’бһҖбһҫбһҸбһңбҹҒбһ‘бһ·бһҖбһ¶бһҠбҹӮбһӣбһҡбһ№бһ„бһҳбһ¶бҹҶ бһ“бһ·бһ„бһҠбҹҶбһҺбһҫбһҡбһҖбһ¶бһҡбһўбһ“бһ»бһӣбҹ„бһҳбҹ”

## бһӮбҹ„бһӣбһ”бҹҶбһҺбһ„бһҹбһ·бһҖбҹ’бһҹбһ¶

бһ“бҹ…бһ…бһ»бһ„бһ”бһ„бҹ’бһўбһҹбҹӢбһ“бҹғбһҳбһјбһӣбһҠбҹ’бһӢбһ¶бһ“бһ“бҹҒбҹҮ бһўбҹ’бһ“бһҖбһ“бһ№бһ„бһўбһ¶бһ…бһ’бҹ’бһңбһҫбһ”бһ¶бһ“бҹ–

- **бһўбһ“бһ»бһңбһҸбҹ’бһҸ** бһҹбһ»бһңбһҸбҹ’бһҗбһ·бһ—бһ¶бһ–бһҖбһҳбҹ’бһҡбһ·бһҸбһҮбһҪбһҡбһҠбһ‘бҹғбһҗбҹ’бһ“бһ¶бһҖбҹӢбһҹбһ бһӮбҹ’бһҡбһ¶бһҹбһҹбһҳбҹ’бһҡбһ¶бһ”бҹӢбһҖбһ¶бһҡбһ”бҹҶбһ”бҹӮбһҖбһ‘бһ·бһ“бҹ’бһ“бһ“бҹҗбһҷбһ…бҹ’бһҡбһҫбһ“бһўбһҸбһ·бһҗбһ·бһҮбһ“  
- **бһҡбһ…бһ“бһ¶** бһӣбҹҶбһ“бһ¶бҹҶбһ•бҹ’бһ‘бҹҖбһ„бһ•бҹ’бһ‘бһ¶бһҸбҹӢбһўбһҸбҹ’бһҸбһҹбһүбҹ’бһүбһ¶бһҺ бһ“бһ·бһ„бһўбһ“бһ»бһүбҹ’бһүбһ¶бһҸбһҖбһҳбҹ’бһҳбһҠбҹӮбһӣбһҳбһ¶бһ“бһҹбһ»бһңбһҸбҹ’бһҗбһ·бһ—бһ¶бһ–бһҮбһ¶бһҳбһҪбһҷ Azure  
- **бһҖбҹҶбһҺбһҸбҹӢбһҡбһ…бһ“бһ¶бһҹбһҳбҹ’бһ–бҹҗбһ“бҹ’бһ’** бһҖбҹҶбһҺбһҸбҹӢбһ бҹҒбһҸбһ»бһҸбҹ’бһҡбһҪбһҸбһ–бһ·бһ“бһ·бһҸбҹ’бһҷбһ‘бһјбһӣбҹҶбһ‘бһјбһӣбһ¶бһҷбһҹбһҳбҹ’бһҡбһ¶бһ”бҹӢбһҸбһҳбҹ’бһҡбһјбһңбһҖбһ¶бһҡбһўбһ“бһ»бһӣбҹ„бһҳ  
- **бһўбһ“бһ»бһңбһҸбҹ’бһҸ** бһҷбһ»бһ‘бҹ’бһ’бһҹбһ¶бһҹбҹ’бһҸбҹ’бһҡбһҖбһ¶бһҡбһ–бһ¶бһҡбһҮбһҳбҹ’бһҡбҹҮбһҮбҹ’бһҡбҹ…бһ“бҹ…бһӣбһҫбһҹбҹ’бһҡбһ‘бһ¶бһ”бҹӢбһҖбһҳбҹ’бһҳбһңбһ·бһ’бһёбһ‘бһ¶бҹҶбһ„бһўбһҹбҹӢ  
- **бһ•бҹ’бһ‘бҹҖбһ„бһ•бҹ’бһ‘бһ¶бһҸбҹӢ** бһҖбһ¶бһҡбһўбһ“бһ»бһңбһҸбҹ’бһҸбһҹбһ»бһңбһҸбҹ’бһҗбһ·бһ—бһ¶бһ–бһҸбһ¶бһҳбһҡбһҷбҹҲбһҖбһ¶бһҡбһ’бҹ’бһңбһҫбһҸбҹҒбһҹбҹ’бһҸбһ”бҹ’бһҡбһ–бҹҗбһ“бҹ’бһ’  
- **бһҸбҹ’бһҡбһҪбһҸбһ–бһ·бһ“бһ·бһҸбҹ’бһҷ** бһ–бҹ’бһҡбһ№бһҸбҹ’бһҸбһ·бһҖбһ¶бһҡбһҺбҹҚбһҹбһ»бһңбһҸбҹ’бһҗбһ·бһ—бһ¶бһ– бһ“бһ·бһ„бһҶбҹ’бһӣбһҫбһҷбһҸбһ”бһ…бҹҶбһ–бҹ„бҹҮбһӮбҹ’бһҡбҹ„бҹҮбһҗбҹ’бһ“бһ¶бһҖбҹӢбһўбһ¶бһ…бһҖбһҫбһҸбһҳбһ¶бһ“

```sql
-- Function to securely set store context
CREATE OR REPLACE FUNCTION retail.set_store_context(store_id_param VARCHAR(50))
RETURNS void
LANGUAGE plpgsql
SECURITY DEFINER
SET search_path = retail, pg_temp
AS $$
DECLARE
    user_info RECORD;
BEGIN
    -- Validate store exists and is active
    SELECT store_id, store_name, is_active 
    INTO user_info
    FROM retail.stores 
    WHERE store_id = store_id_param;
    
    IF NOT FOUND THEN
        RAISE EXCEPTION 'Store not found: %', store_id_param
            USING ERRCODE = 'invalid_parameter_value',
                  HINT = 'Verify store ID and ensure it exists in the system';
    END IF;
    
    IF NOT user_info.is_active THEN
        RAISE EXCEPTION 'Store is inactive: %', store_id_param
            USING ERRCODE = 'insufficient_privilege',
                  HINT = 'Contact administrator to activate store';
    END IF;
    
    -- Set the secure context
    PERFORM set_config('app.current_store_id', store_id_param, false);
    PERFORM set_config('app.store_name', user_info.store_name, false);
    PERFORM set_config('app.context_set_at', extract(epoch from current_timestamp)::text, false);
    
    -- Log context change for audit
    INSERT INTO retail.security_audit_log (
        event_type,
        user_name,
        store_id,
        ip_address,
        user_agent,
        details,
        severity
    ) VALUES (
        'store_context_set',
        current_user,
        store_id_param,
        inet_client_addr()::text,
        current_setting('application_name', true),
        jsonb_build_object(
            'store_name', user_info.store_name,
            'timestamp', current_timestamp,
            'session_id', pg_backend_pid()
        ),
        'INFO'
    );
END;
$$;

-- Grant execute to MCP user
GRANT EXECUTE ON FUNCTION retail.set_store_context TO mcp_user;
```

### бһҖбһ¶бһҡбһҡбһҪбһҳбһ”бһүбҹ’бһ…бһјбһӣ Azure Entra ID

```python
# mcp_server/security/authentication.py
"""
Azure Entra ID authentication for MCP server.
"""
import os
import jwt
import aiohttp
import asyncio
from typing import Dict, Optional, List
from datetime import datetime, timezone
from azure.identity.aio import DefaultAzureCredential
from azure.keyvault.secrets.aio import SecretClient
import logging

logger = logging.getLogger(__name__)

class AzureAuthenticator:
    """Handle Azure Entra ID authentication and token validation."""
    
    def __init__(self):
        self.tenant_id = os.getenv('AZURE_TENANT_ID')
        self.client_id = os.getenv('AZURE_CLIENT_ID')
        self.audience = os.getenv('AZURE_AUDIENCE', self.client_id)
        self.issuer = f"https://login.microsoftonline.com/{self.tenant_id}/v2.0"
        
        # бһҖбҹӮбһ…бҹӢбһҹбһҳбҹ’бһҡбһ¶бһ”бҹӢ JWKS (JSON Web Key Set)
        self._jwks_cache = None
        self._jwks_cache_expiry = None
        
        # бһҸбһ»бһҡбһҖбһҳбҹ’бһҳбһҹбһҳбҹ’бһҡбһ¶бһ”бҹӢбһҹбһҳбҹ’бһ„бһ¶бһҸбҹӢ
        self.key_vault_url = os.getenv('AZURE_KEY_VAULT_URL')
        self.credential = DefaultAzureCredential()
        
        if self.key_vault_url:
            self.secret_client = SecretClient(
                vault_url=self.key_vault_url,
                credential=self.credential
            )
    
    async def validate_token(self, token: str) -> Dict:
        """Validate JWT token from Azure Entra ID."""
        
        try:
            # бһ‘бһ‘бһҪбһӣбһ”бһ¶бһ“бһҖбһјбһ“бһҹбҹ„бһҹбһҳбҹ’бһҡбһ¶бһ”бҹӢбһ…бһ»бҹҮбһ бһҸбҹ’бһҗбһӣбҹҒбһҒбһ¶
            signing_keys = await self._get_signing_keys()
            
            # бһңбһ¶бһҷбһ”бһҖбһҒбҹ’бһ“бһ„бһҸбһјбһҠбҹӮбһӣбһҠбһҫбһҳбҹ’бһ”бһёбһ‘бһ‘бһҪбһӣбһ”бһ¶бһ“бһўбһҸбҹ’бһҸбһҹбһүбҹ’бһүбһ¶бһҺбһҖбһјбһ“бһҹбҹ„
            unverified_header = jwt.get_unverified_header(token)
            key_id = unverified_header.get('kid')
            
            if not key_id:
                raise ValueError("Token missing key ID")
            
            # бһҹбҹ’бһңбҹӮбһ„бһҡбһҖбһҖбһјбһ“бһҹбҹ„бһҠбҹӮбһӣбһҹбһҳбһҡбһҳбҹ’бһҷ
            signing_key = None
            for key in signing_keys:
                if key['kid'] == key_id:
                    signing_key = jwt.algorithms.RSAAlgorithm.from_jwk(key)
                    break
            
            if not signing_key:
                raise ValueError(f"Unable to find signing key for kid: {key_id}")
            
            # бһ•бҹ’бһ‘бҹҖбһ„бһ•бҹ’бһ‘бһ¶бһҸбҹӢ бһ“бһ·бһ„бһңбһ¶бһҷбһ”бһҖбһҸбһјбһҠбҹӮбһӣ
            payload = jwt.decode(
                token,
                signing_key,
                algorithms=['RS256'],
                audience=self.audience,
                issuer=self.issuer,
                options={
                    'verify_exp': True,
                    'verify_aud': True,
                    'verify_iss': True
                }
            )
            
            # бһҠбһҖбһҷбһҖбһ–бҹҗбһҸбҹҢбһҳбһ¶бһ“бһўбҹ’бһ“бһҖбһ”бҹ’бһҡбһҫ
            user_info = self._extract_user_info(payload)
            
            # бһҖбһҸбҹӢбһҸбҹ’бһҡбһ¶бһҖбһ¶бһҡбһ•бҹ’бһ‘бҹҖбһ„бһ•бҹ’бһ‘бһ¶бһҸбҹӢбһҮбҹ„бһӮбһҮбҹҗбһҷ
            logger.info(
                "User authenticated successfully",
                extra={
                    'user_id': user_info['user_id'],
                    'email': user_info.get('email'),
                    'tenant_id': payload.get('tid')
                }
            )
            
            return user_info
            
        except jwt.ExpiredSignatureError:
            logger.warning("Token has expired")
            raise ValueError("Token has expired")
        except jwt.InvalidAudienceError:
            logger.warning(f"Invalid audience in token. Expected: {self.audience}")
            raise ValueError("Invalid token audience")
        except jwt.InvalidIssuerError:
            logger.warning(f"Invalid issuer in token. Expected: {self.issuer}")
            raise ValueError("Invalid token issuer")
        except Exception as e:
            logger.error(f"Token validation failed: {str(e)}")
            raise ValueError(f"Token validation failed: {str(e)}")
    
    async def _get_signing_keys(self) -> List[Dict]:
        """Get JWKS from Azure Entra ID with caching."""
        
        current_time = datetime.now(timezone.utc)
        
        # бһ–бһ·бһ“бһ·бһҸбҹ’бһҷбһҳбһҫбһӣбһҗбһ¶бһӮбҹӮбһ…бҹӢбһҸбҹ’бһҡбһ№бһҳбһҸбҹ’бһҡбһјбһңбһҡбһәбһ‘бҹҒ
        if (self._jwks_cache and self._jwks_cache_expiry and 
            current_time < self._jwks_cache_expiry):
            return self._jwks_cache
        
        # бһ‘бһ¶бһүбһҷбһҖ JWKS бһҗбҹ’бһҳбһё
        jwks_url = f"{self.issuer}/keys"
        
        async with aiohttp.ClientSession() as session:
            async with session.get(jwks_url) as response:
                if response.status != 200:
                    raise Exception(f"Failed to fetch JWKS: {response.status}")
                
                jwks_data = await response.json()
                
        # бһҖбҹӮбһ…бҹӢбһҹбһҳбҹ’бһҡбһ¶бһ”бҹӢбһҳбһҪбһҷбһҳбҹүбҹ„бһ„
        self._jwks_cache = jwks_data['keys']
        self._jwks_cache_expiry = current_time.replace(
            hour=current_time.hour + 1
        )
        
        return self._jwks_cache
    
    def _extract_user_info(self, payload: Dict) -> Dict:
        """Extract user information from JWT payload."""
        
        return {
            'user_id': payload.get('oid') or payload.get('sub'),
            'email': payload.get('email') or payload.get('preferred_username'),
            'name': payload.get('name'),
            'tenant_id': payload.get('tid'),
            'roles': payload.get('roles', []),
            'groups': payload.get('groups', []),
            'app_roles': payload.get('app_roles', []),
            'scope': payload.get('scp', '').split() if payload.get('scp') else [],
            'expires_at': datetime.fromtimestamp(payload['exp'], timezone.utc),
            'issued_at': datetime.fromtimestamp(payload['iat'], timezone.utc)
        }
    
    async def get_user_store_access(self, user_id: str) -> List[str]:
        """Get list of stores the user has access to."""
        
        try:
            # бһ“бҹҒбҹҮбһҮбһ¶бһ‘бһјбһ‘бҹ…бһҠбһҫбһҳбҹ’бһ”бһёбһҹбһҪбһҡбһҹбҹҶбһҺбһҪбһҡбһ–бһёбһҖбһ¶бһҡбһ…бһ„бһҖбҹ’бһҡбһ„бһўбҹ’бһ“бһҖбһ”бҹ’бһҡбһҫ/бһ бһ¶бһ„бһҡбһ”бһҹбҹӢбһўбҹ’бһ“бһҖ
            # бһҹбһҳбҹ’бһҡбһ¶бһ”бҹӢбһҖбһҡбһҺбһёбһҹбһ¶бһҖбһӣбҹ’бһ”бһ„ бһҷбһҫбһ„бһ“бһ№бһ„бһ”бҹ’бһҡбһҫбһҹбһҳбҹ’бһ„бһ¶бһҸбҹӢбһҸбһјбһ…бһҳбһҪбһҷбһҖбҹ’бһ“бһ»бһ„бһҸбһ»бһҡбһҖбһҳбҹ’бһҳбһҹбҹ„
            secret_name = f"user-{user_id}-stores"
            
            if self.secret_client:
                secret = await self.secret_client.get_secret(secret_name)
                store_list = secret.value.split(',')
                return [store.strip() for store in store_list if store.strip()]
            
            # бһҮбһҳбҹ’бһҡбһҫбһҹбһ”бһүбҹ’бһ…бҹҒбһү: бһҸбҹ’бһҡбһЎбһ”бҹӢбһҖбһ¶бһҡбһ…бһјбһӣбһ бһ¶бһ„бһӣбҹҶбһ“бһ¶бҹҶбһҠбһҫбһҳ
            logger.warning(f"No store mapping found for user {user_id}, using default")
```

```python
# mcp_server/security/authorization.py
"""
Authorization middleware and decorators for MCP server.
"""
import functools
from typing import Dict, List, Optional, Callable, Any
from fastapi import HTTPException, status, Request
from fastapi.security import HTTPBearer, HTTPAuthorizationCredentials
import logging

logger = logging.getLogger(__name__)

security = HTTPBearer()

class AuthorizationError(Exception):

class RoleBasedAuth:
    """Role-based access control implementation."""
    
    # бһҖбҹҶбһҺбһҸбҹӢбһҮбһҪбһҡбһҠбһ№бһҖбһ“бһ¶бҹҶбһҸбһҪбһ“бһ¶бһ‘бһё
    ROLE_HIERARCHY = {
        'store_admin': ['store_manager', 'store_user', 'store_readonly'],
        'store_manager': ['store_user', 'store_readonly'],
        'store_user': ['store_readonly'],
        'store_readonly': []
    }
    
    # бһҖбҹҶбһҺбһҸбҹӢбһҹбһ·бһ‘бҹ’бһ’бһ·бһҹбһҳбҹ’бһҡбһ¶бһ”бҹӢбһҸбһҪбһ“бһ¶бһ‘бһёбһҳбһҪбһҷбҹ—
    ROLE_PERMISSIONS = {
        'store_admin': [
            'read_all', 'write_all', 'delete_all', 'manage_users'
        ],
        'store_manager': [
            'read_all', 'write_transactions', 'write_inventory', 'read_reports'
        ],
        'store_user': [
            'read_products', 'read_customers', 'write_transactions'
        ],
        'store_readonly': [
            'read_products', 'read_basic_reports'
        ]
    }
    
    @classmethod
    def has_permission(cls, user_roles: List[str], required_permission: str) -> bool:
        """Check if user has required permission."""
        
        user_permissions = set()
        
        for role in user_roles:
            # бһ”бһ“бҹ’бһҗбҹӮбһҳбһҹбһ·бһ‘бһ·бҹ’бһ’бһҠбҹ„бһҷбһ•бҹ’бһ‘бһ¶бһӣбҹӢ
            user_permissions.update(cls.ROLE_PERMISSIONS.get(role, []))
            
            # бһ”бһ“бҹ’бһҗбҹӮбһҳбһҹбһ·бһ‘бһ·бҹ’бһ’бһҠбҹӮбһӣбһ‘бһ‘бһҪбһӣбһ”бһ¶бһ“бһ–бһёбһҳбҹ’бһҸбһ¶бһҷ
            inherited_roles = cls.ROLE_HIERARCHY.get(role, [])
            for inherited_role in inherited_roles:
                user_permissions.update(cls.ROLE_PERMISSIONS.get(inherited_role, []))
        
        return required_permission in user_permissions
    
    @classmethod
    def get_user_stores(cls, user_info: Dict) -> List[str]:
        """Extract stores user has access to from user info."""
        
        # бһ“бҹҒбҹҮбһ’бһҳбҹ’бһҳбһҸбһ¶бһ“бһ№бһ„бһҳбһҖбһ–бһёбһ”бҹ’бһҡбһ–бҹҗбһ“бҹ’бһ’бһӮбҹ’бһҡбһ”бҹӢбһӮбҹ’бһҡбһ„бһўбҹ’бһ“бһҖбһ”бҹ’бһҡбһҫбһҡбһ”бһҹбҹӢбһўбҹ’бһ“бһҖ
        # бһҹбһҳбҹ’бһҡбһ¶бһ”бҹӢбһҹбһ¶бһҖбһӣбҹ’бһ”бһ„ бһҷбһҫбһ„бһ“бһ№бһ„бһҠбһҖбһҹбҹ’бһҡбһ„бҹӢбһ–бһёбһҖбһ¶бһҡбһўбҹҮбһўбһ¶бһ„бһӣбҹҶбһ“бһ¶бҹҶбһ•бҹ’бһ‘бһ¶бһӣбҹӢбһҒбҹ’бһӣбһҪбһ“бһ¬бһҖбҹ’бһҡбһ»бһҳ
        
        stores = []
        
        # бһ–бһ·бһ“бһ·бһҸбҹ’бһҷбһҖбһ¶бһҡбһ…бһ¶бһҸбҹӢбһ…бҹӮбһ„бһ бһ¶бһ„бһҠбҹ„бһҷбһ•бҹ’бһ‘бһ¶бһӣбҹӢбһҖбҹ’бһ“бһ»бһ„бһҖбҹ’бһҡбһ»бһҳ
        for group in user_info.get('groups', []):
            if group.startswith('store_'):
                store_id = group.replace('store_', '')
                stores.append(store_id)
        
        # бһ–бһ·бһ“бһ·бһҸбҹ’бһҷбһҸбһҪбһ“бһ¶бһ‘бһёбһ–бһ¶бһҖбҹӢбһ–бҹҗбһ“бҹ’бһ’бһҖбһҳбҹ’бһҳбһңбһ·бһ’бһё
        for role in user_info.get('app_roles', []):
            if 'store:' in role:
                _, store_id = role.split('store:', 1)
                stores.append(store_id)
        
        return list(set(stores))  # бһӣбһ»бһ”бһ…бҹ„бһӣбһҖбһ¶бһҡбһ…бһҳбҹ’бһӣбһ„

def require_auth(required_permission: str = None, require_store_access: bool = True):
    """Decorator to require authentication and authorization."""
    
    def decorator(func: Callable) -> Callable:
        @functools.wraps(func)
        async def wrapper(*args, **kwargs):
            # бһҠбһҖбһҷбһҖбһҖбһ¶бһҡбһҹбҹ’бһ“бһҫбһҹбһ»бҹҶбһ–бһё args (бһҖбһ¶бһҡInjectбһ–бһ№бһ„бһ•бҹ’бһўбҹӮбһҖFastAPI)
            request = None
            for arg in args:
                if isinstance(arg, Request):
                    request = arg
                    break
            
            if not request:
                raise HTTPException(
                    status_code=status.HTTP_500_INTERNAL_SERVER_ERROR,
                    detail="Request object not found"
                )
            
            # бһ‘бһ‘бһҪбһӣбһ”бһ¶бһ“бһҖбҹ’бһ”бһ¶бһӣбһўбһҸбһ·бһҗбһ·бһҮбһ“бһҹбһҳбҹ’бһ„бһ¶бһҸбҹӢ
            auth_header = request.headers.get('Authorization')
            if not auth_header or not auth_header.startswith('Bearer '):
                raise HTTPException(
                    status_code=status.HTTP_401_UNAUTHORIZED,
                    detail="Missing or invalid authorization header",
                    headers={"WWW-Authenticate": "Bearer"}
                )
            
            token = auth_header.split(' ')[1]
            
            try:
                # бһҸбҹ’бһҡбһҪбһҸбһ–бһ·бһ“бһ·бһҸбҹ’бһҷбһҹбһүбҹ’бһүбһ¶бһ”бһҸбҹ’бһҡ
                user_info = await azure_authenticator.validate_token(token)
                
                # бһ–бһ·бһ“бһ·бһҸбҹ’бһҷбһҹбһ·бһ‘бҹ’бһ’бһ·бһ…бһ¶бҹҶбһ”бһ¶бһ…бҹӢ
                if required_permission:
                    user_roles = user_info.get('roles', [])
                    if not RoleBasedAuth.has_permission(user_roles, required_permission):
                        raise HTTPException(
                            status_code=status.HTTP_403_FORBIDDEN,
                            detail=f"Insufficient permissions. Required: {required_permission}"
                        )
                
                # бһ–бһ·бһ“бһ·бһҸбҹ’бһҷбһҖбһ¶бһҡбһ…бһјбһӣбһҠбҹҶбһҺбһҫбһҡбһҖбһ¶бһҡбһ бһ¶бһ„
                if require_store_access:
                    user_stores = RoleBasedAuth.get_user_stores(user_info)
                    if not user_stores:
                        raise HTTPException(
                            status_code=status.HTTP_403_FORBIDDEN,
                            detail="No store access configured for user"
                        )
                    
                    # бһҖбҹҶбһҺбһҸбҹӢбһ”бһҡбһ·бһ”бһ‘бһ бһ¶бһ„бһӣбҹҶбһ“бһ¶бҹҶбһҠбһҫбһҳ (бһ бһ¶бһ„бһҠбҹҶбһ”бһјбһ„бһҠбҹӮбһӣбһ…бһјбһӣбһҠбҹҶбһҺбһҫбһҡбһҖбһ¶бһҡ)
                    request.state.current_store = user_stores[0]
                    request.state.accessible_stores = user_stores
                
                # бһ”бһ“бҹ’бһҗбҹӮбһҳбһ–бҹҗбһҸбҹҢбһҳбһ¶бһ“бһўбҹ’бһ“бһҖбһ”бҹ’бһҡбһҫбһ‘бҹ…бһҹбҹ’бһҗбһ¶бһ“бһ—бһ¶бһ–бһҹбҹ’бһ“бһҫбһҹбһ»бҹҶ
                request.state.user_info = user_info
                request.state.user_id = user_info['user_id']
                
                # бһ бҹ…бһҳбһ»бһҒбһ„бһ¶бһҡбһҠбһҫбһҳ
                return await func(*args, **kwargs)
                
            except ValueError as e:
                raise HTTPException(
                    status_code=status.HTTP_401_UNAUTHORIZED,
                    detail=str(e),
                    headers={"WWW-Authenticate": "Bearer"}
                )
            except AuthorizationError as e:
                raise HTTPException(
                    status_code=status.HTTP_403_FORBIDDEN,
                    detail=str(e)
                )
        
        return wrapper
    return decorator

def require_store_context(store_param: str = 'store_id'):
    """Decorator to validate and set store context."""
    
    def decorator(func: Callable) -> Callable:
        @functools.wraps(func)
        async def wrapper(*args, **kwargs):
            # бһ‘бһ‘бһҪбһӣ store_id бһ–бһё kwargs
            store_id = kwargs.get(store_param)
            
            if not store_id:
                raise HTTPException(
                    status_code=status.HTTP_400_BAD_REQUEST,
                    detail=f"Missing required parameter: {store_param}"
                )
            
            # бһҠбһҖбһҹбҹ’бһҡбһ„бҹӢбһҹбҹҶбһҺбһҫбһ–бһё args
            request = None
            for arg in args:
                if isinstance(arg, Request):
                    request = arg
                    break
            
            if not request or not hasattr(request.state, 'accessible_stores'):
                raise HTTPException(
                    status_code=status.HTTP_500_INTERNAL_SERVER_ERROR,
                    detail="Authentication required before store context validation"
                )
            
            # бһҸбҹ’бһҡбһҪбһҸбһ–бһ·бһ“бһ·бһҸбҹ’бһҷбһўбҹ’бһ“бһҖбһ”бҹ’бһҡбһҫбһҳбһ¶бһ“бһҖбһ¶бһҡбһ…бһјбһӣбһҠбҹҶбһҺбһҫбһҡбһҖбһ¶бһҡбһ бһ¶бһ„бһҠбҹӮбһӣбһ”бһ¶бһ“бһҹбҹ’бһ“бһҫбһҹбһ»бҹҶбһҠбҹӮбһҡбһ¬бһ‘бҹҒ
            if store_id not in request.state.accessible_stores:
                raise HTTPException(
                    status_code=status.HTTP_403_FORBIDDEN,
                    detail=f"Access denied to store: {store_id}"
                )
            
            # бһҖбҹҶбһҺбһҸбҹӢбһ”бһҡбһ·бһ”бһ‘бһ бһ¶бһ„бһ“бҹ…бһҖбҹ’бһ“бһ»бһ„бһҹбҹ’бһҗбһ¶бһ“бһ—бһ¶бһ–бһҹбҹ’бһ“бһҫбһҹбһ»бҹҶ
            request.state.current_store = store_id
            
            return await func(*args, **kwargs)
        
        return wrapper
    return decorator
```

```sql
CREATE VIEW retail.security_failed_auth AS
SELECT 
    event_type,
    user_name,
    ip_address,
    COUNT(*) as attempt_count,
    MIN(created_at) as first_attempt,
    MAX(created_at) as last_attempt,
    ARRAY_AGG(DISTINCT failure_reason) as failure_reasons
FROM retail.security_audit_log
WHERE success = FALSE 
  AND event_type IN ('authentication_failed', 'token_validation_failed')
  AND created_at >= CURRENT_TIMESTAMP - INTERVAL '24 hours'
GROUP BY event_type, user_name, ip_address
HAVING COUNT(*) >= 3  -- 3 or more failures
ORDER BY attempt_count DESC, last_attempt DESC;

CREATE VIEW retail.security_suspicious_access AS
SELECT 
    user_name,
    user_id,
    COUNT(DISTINCT ip_address) as ip_count,
    COUNT(DISTINCT store_id) as store_count,
    ARRAY_AGG(DISTINCT ip_address::TEXT) as ip_addresses,
    ARRAY_AGG(DISTINCT store_id) as stores_accessed,
    MIN(created_at) as first_access,
    MAX(created_at) as last_access
FROM retail.security_audit_log
WHERE created_at >= CURRENT_TIMESTAMP - INTERVAL '1 hour'
  AND success = TRUE
GROUP BY user_name, user_id
HAVING COUNT(DISTINCT ip_address) > 3  -- Access from multiple IPs
   OR COUNT(DISTINCT store_id) > 2     -- Access to multiple stores
ORDER BY ip_count DESC, store_count DESC;

    store_id,
    resource_type,
    action,

    COUNT(DISTINCT user_id) as unique_users

WHERE resource_type IS NOT NULL
  AND created_at >= CURRENT_TIMESTAMP - INTERVAL '24 hours'
```

```python
        self.thresholds = {
            'failed_auth_attempts': 5,      # бһҸбһ¶бһҳбһўбҹ’бһ“бһҖбһ”бҹ’бһҡбһҫбһ”бҹ’бһҡбһ¶бһҹбҹӢбһҳбһҪбһҷбһҳбҹ’бһ“бһ¶бһҖбҹӢбһҖбҹ’бһ“бһ»бһ„бһҳбһҪбһҷбһҳбҹүбҹ„бһ„
            'multiple_ip_access': 3,        # бһўбһ¶бһҹбһҷбһҠбҹ’бһӢбһ¶бһ“ IP бһ•бҹ’бһҹбҹҒбһ„бҹ—бһҸбһ¶бһҳбһўбҹ’бһ“бһҖбһ”бҹ’бһҡбһҫбһ”бҹ’бһҡбһ¶бһҹбҹӢбһҳбһҪбһҷбһҖбҹ’бһ“бһ»бһ„бһҳбһҪбһҷбһҳбҹүбҹ„бһ„
            'excessive_data_access': 1000,  # бһҹбҹҶбһҺбһҪбһҡбһҸбһ¶бһҳбһўбҹ’бһ“бһҖбһ”бҹ’бһҡбһҫбһ”бҹ’бһҡбһ¶бһҹбҹӢбһҳбһҪбһҷбһҖбҹ’бһ“бһ»бһ„бһҳбһҪбһҷбһҳбҹүбҹ„бһ„
            'privilege_escalation': 1,      # бһҖбһ¶бһҡбһ§бҹ’бһҷбһ¶бһҷбһ¶бһҳбһҺбһ¶бһҳбһҪбһҷ
            'unauthorized_store_access': 1  # бһҖбһ¶бһҡбһ§бҹ’бһҷбһ¶бһҷбһ¶бһҳбһҺбһ¶бһҳбһҪбһҷ
        }
    
    async def start_monitoring(self):
        """Start security monitoring loop."""
        logger.info("Starting security monitoring")
        
        while True:
            try:
                await self._check_security_events()
                await asyncio.sleep(300)  # бһ–бһ·бһ“бһ·бһҸбҹ’бһҷбһҡбҹҖбһ„бһҡбһ¶бһӣбҹӢ бҹҘ бһ“бһ¶бһ‘бһё
            except Exception as e:
                logger.error(f"Security monitoring error: {e}")
                await asyncio.sleep(60)  # бһҹбһ¶бһҖбһӣбҹ’бһ”бһ„бһЎбһҫбһ„бһңбһ·бһүбһҷбҹүбһ¶бһ„бһҒбҹ’бһӣбһёбһ“бҹ…бһ–бҹҒбһӣбһҳбһ¶бһ“бһҖбҹҶбһ бһ»бһҹ
    
    async def _check_security_events(self):
        """Check for security events and generate alerts."""
        
        conn = await asyncpg.connect(self.db_connection_string)
        
        try:
            # бһ–бһ·бһ“бһ·бһҸбҹ’бһҷбһҖбһ¶бһҡбһ§бҹ’бһҷбһ¶бһҷбһ¶бһҳбһўбҹҮбһўбһ¶бһ„бһҖбҹҶбһҺбһҸбҹӢбһўбһҸбҹ’бһҸбһҹбһүбҹ’бһүбһ¶бһҺбһҠбҹӮбһӣбһ”бһҡбһ¶бһҮбҹҗбһҷ
            await self._check_failed_auth(conn)
            
            # бһ–бһ·бһ“бһ·бһҸбҹ’бһҷбһӣбҹҶбһ“бһ¶бҹҶбһ…бһјбһӣбһҠбҹӮбһӣбһӮбһҪбһҡбһІбҹ’бһҷбһҹбһ„бһ„бһ№бһҸ
            await self._check_suspicious_access(conn)
            
            # бһ–бһ·бһ“бһ·бһҸбҹ’бһҷбһўбһ¶бһҹбһҷбһҠбҹ’бһӢбһ¶бһ“бһ‘бһ·бһ“бҹ’бһ“бһ“бҹҗбһҷбһҠбҹӮбһӣбһҳбһ¶бһ“бһ—бһ¶бһ–бһ…бҹ’бһҡбһ”бһјбһҖбһ…бҹ’бһҡбһ”бһӣбҹӢ
            await self._check_data_access_anomalies(conn)
            
            # бһ–бһ·бһ“бһ·бһҸбҹ’бһҷбһҖбһ¶бһҡбһ§бҹ’бһҷбһ¶бһҷбһ¶бһҳбһ…бһјбһӣбһҠбҹ„бһҷбһӮбҹ’бһҳбһ¶бһ“бһҹбһ·бһ‘бҹ’бһ’бһ·
            await self._check_unauthorized_access(conn)
            
        finally:
            await conn.close()
    
    async def _check_failed_auth(self, conn):
        """Check for excessive failed authentication attempts."""
        
        query = """
        SELECT 
            user_name,
            ip_address,
            COUNT(*) as attempt_count,
            MAX(created_at) as last_attempt
        FROM retail.security_audit_log
        WHERE success = FALSE 
          AND event_type IN ('authentication_failed', 'token_validation_failed')
          AND created_at >= CURRENT_TIMESTAMP - INTERVAL '1 hour'
        GROUP BY user_name, ip_address
        HAVING COUNT(*) >= $1
        """
        
        results = await conn.fetch(query, self.thresholds['failed_auth_attempts'])
        
        for record in results:
            alert = SecurityAlert(
                alert_type='failed_authentication',
                severity='HIGH',
                message=f"Excessive failed login attempts for user {record['user_name']}",
                details={
                    'user_name': record['user_name'],
                    'ip_address': str(record['ip_address']),
                    'attempt_count': record['attempt_count'],
                    'last_attempt': record['last_attempt'].isoformat()
                },
                timestamp=datetime.now()
            )
            
            await self._send_alert(alert)
    
    async def _check_suspicious_access(self, conn):
        """Check for suspicious access patterns."""
        
        query = """
        SELECT 
            user_name,
            user_id,
            COUNT(DISTINCT ip_address) as ip_count,
            ARRAY_AGG(DISTINCT ip_address::TEXT) as ip_addresses
        FROM retail.security_audit_log
        WHERE created_at >= CURRENT_TIMESTAMP - INTERVAL '1 hour'
          AND success = TRUE
        GROUP BY user_name, user_id
        HAVING COUNT(DISTINCT ip_address) >= $1
        """
        
        results = await conn.fetch(query, self.thresholds['multiple_ip_access'])
        
        for record in results:
            alert = SecurityAlert(
                alert_type='suspicious_access',
                severity='MEDIUM',
                message=f"User {record['user_name']} accessed from multiple IP addresses",
                details={
                    'user_name': record['user_name'],
                    'user_id': record['user_id'],
                    'ip_count': record['ip_count'],
                    'ip_addresses': record['ip_addresses']
                },
                timestamp=datetime.now()
            )
            
            await self._send_alert(alert)
    
    async def _check_unauthorized_access(self, conn):
        """Check for unauthorized store access attempts."""
        
        query = """
        SELECT 
            user_name,
            user_id,
            store_id,
            failure_reason,
            created_at
        FROM retail.security_audit_log
        WHERE success = FALSE 
          AND event_type = 'unauthorized_store_access'
          AND created_at >= CURRENT_TIMESTAMP - INTERVAL '1 hour'
        """
        
        results = await conn.fetch(query)
        
        for record in results:
            alert = SecurityAlert(
                alert_type='unauthorized_access',
                severity='HIGH',
                message=f"Unauthorized store access attempt by {record['user_name']}",
                details={
                    'user_name': record['user_name'],
                    'user_id': record['user_id'],
                    'store_id': record['store_id'],
                    'failure_reason': record['failure_reason'],
                    'timestamp': record['created_at'].isoformat()
                },
                timestamp=datetime.now()
            )
            
            await self._send_alert(alert)
    
    async def _send_alert(self, alert: SecurityAlert):
        """Send security alert to all configured handlers."""
        
        logger.warning(
            f"Security Alert: {alert.alert_type} - {alert.message}",
            extra={'alert_details': alert.details}
        )
        
        # аёӘа№ҲаёҮбһ‘бҹ…бһҠбһӣбҹӢбһўбҹ’бһ“бһҖбһ‘бһ‘бһҪбһӣбһҮбһјбһ“бһҹбһ¶бһҡбһ–бҹ’бһҡбһҳбһ¶бһ“бһҠбҹӮбһӣбһ”бһ¶бһ“бһҖбҹҶбһҺбһҸбҹӢ
        for handler in self.alert_handlers:
            try:
                await handler.send_alert(alert)
            except Exception as e:
                logger.error(f"Failed to send alert via {handler.__class__.__name__}: {e}")
    
    def add_alert_handler(self, handler):
        """Add alert handler."""
        self.alert_handlers.append(handler)
```
