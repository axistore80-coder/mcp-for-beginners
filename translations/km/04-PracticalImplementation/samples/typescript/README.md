# бһ§бһ‘бһ¶бһ бһҡбһҺбҹҚ

бһ“бҹҒбҹҮбһӮбһәбһҮбһ¶бһ§бһ‘бһ¶бһ бһҡбһҺбҹҚ Typescript бһҹбһҳбҹ’бһҡбһ¶бһ”бҹӢбһҳбҹүбһ¶бһҹбҹҠбһёбһ“бһҳбҹҒ MCP

бһ“бҹҒбҹҮбһӮбһәбһҮбһ¶бһӮбҹҶбһҡбһјбһҖбһ¶бһҡбһ”бһ„бҹ’бһҖбһҫбһҸбһ§бһ”бһҖбһҡбһҺбҹҚбҹ–

```typescript
this.mcpServer.tool(
'completion',
{
    model: z.string(),
    prompt: z.string(),
    options: z.object({
        temperature: z.number().optional(),
        max_tokens: z.number().optional(),
        stream: z.boolean().optional()
    }).optional()
},
async ({ model, prompt, options }) => {
    console.log(`Processing completion request for model: ${model}`);

    // бһ”бһүбҹ’бһ…бһ¶бһҖбҹӢбһҳбҹүбһјбһҠбҹӮбһӣ
    if (!this.models.includes(model)) {
        throw new Error(`Model ${model} not supported`);
    }

    // бһ•бҹ’бһӮбһ»бҹҶбһІбҹ’бһҷбһҳбһ¶бһ“бһ–бҹ’бһҡбһ№бһҸбҹ’бһҸбһ·бһҖбһ¶бһҡбһҺбҹҚбһҹбһҳбҹ’бһҡбһ¶бһ”бҹӢбһҸбһ¶бһҳбһҠбһ¶бһ“/бһңбһ¶бһҹбҹӢбһңбҹӮбһ„
    this.events.emit('request', { 
        type: 'completion', 
        model, 
        timestamp: new Date() 
    });

    // бһҖбҹ’бһ“бһ»бһ„бһҖбһ¶бһҡбһўбһ“бһ»бһңбһҸбҹ’бһҸбһ”бҹ’бһҡбһ¶бһҖбһҠ бһ“бҹҒбҹҮбһ“бһ№бһ„бһ бҹ…бһҳбҹүбһјбһҠбҹӮбһӣ AI
    // бһ“бҹ…бһ‘бһёбһ“бҹҒбҹҮ бһҷбһҫбһ„бһӮбҹ’бһҡбһ¶бһ“бҹӢбһҸбҹӮбһ”бҹҶбһӣбҹӮбһ„бһ•бҹ’бһ“бҹӮбһҖбһ“бҹғбһҹбҹҶбһҺбһҫбһҮбһ¶бһҳбһҪбһҷбһ“бһ№бһ„бһҸбһ”бһҹбҹ’бһҳбһ¶бһ“
    const response = {
        id: `mcp-resp-${Date.now()}`,
        model,
        text: `This is a response to: ${prompt.substring(0, 30)}...`,
        usage: {
        promptTokens: prompt.split(' ').length,
        completionTokens: 20,
        totalTokens: prompt.split(' ').length + 20
        }
    };

    // бһ’бҹ’бһңбһҫбһұбҹ’бһҷбһҳбһ¶бһ“бһҖбһ¶бһҡбһ–бһ“бҹ’бһҷбһ¶бһҡбһ–бҹҒбһӣбһ”бһҺбҹ’бһҸбһ¶бһү
    await new Promise(resolve => setTimeout(resolve, 500));

    // бһ•бҹ’бһӮбһ»бҹҶбһІбҹ’бһҷбһҳбһ¶бһ“бһ–бҹ’бһҡбһ№бһҸбҹ’бһҸбһ·бһҖбһ¶бһҡбһҺбҹҚбһ”бһүбҹ’бһ…бһ”бҹӢ
    this.events.emit('completion', {
        model,
        timestamp: new Date()
    });

    return {
        content: [
        {
            type: 'text',
            text: JSON.stringify(response)
        }
        ]
    };
}
);
```
