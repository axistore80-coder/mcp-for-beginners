# бһҖбһ¶бһҡбһҡбһҪбһҳбһ”бһүбҹ’бһ…бһјбһӣбһҳбһҪбһҷбһ…бҹҶбһ“бһҪбһ“

бһҖбһҳбҹ’бһҳбһңбһ·бһ’бһёбһҳбҹүбһјбһҢбҹӮбһӣбһ…бҹ’бһҡбһҫбһ“бһҖбҹҶбһ–бһ»бһ„бһҖбҹ’бһӣбһ¶бһҷбһ‘бҹ…бһҮбһ¶бһҡбһҝбһ„бһҹбҹҶбһҒбһ¶бһ“бҹӢбһҖбһ¶бһ“бҹӢбһҸбҹӮбһ…бҹ’бһҡбһҫбһ“бһ“бҹ…бһҖбҹ’бһ“бһ»бһ„ AI бһҠбҹ„бһҷбһўбһ¶бһ…бһ”бһ„бҹ’бһҖбһҫбһҸбһҖбһ¶бһҡбһ”бҹүбҹҮбһ–бһ¶бһӣбҹӢбһҳбһ¶бһ“бһ—бһ¶бһ–бһҹбҹҶбһ”бһјбһҡбһ”бҹӮбһ” бһ“бһ·бһ„бһ—бһ¶бһҡбһҖбһ·бһ…бҹ’бһ…бһҹбҹ’бһҳбһ»бһӮбһҹбҹ’бһҳбһ¶бһүбһҮбһ¶бһ„бһҳбһ»бһ“бҹ” бһ–бһ·бһ’бһёбһҹбһ¶бһҹбҹ’бһҡбҹ’бһҸ Model Context Protocol (MCP) бһ•бҹ’бһҸбһӣбҹӢбһҹбҹҠбһ»бһҳбһҹбҹҠбһёбһҳбһҹбһҳбҹ’бһҡбһ¶бһ”бҹӢбһҹбҹ’бһҗбһ¶бһ”бһ“бһ¶бһҖбһҳбҹ’бһҳбһңбһ·бһ’бһёбһҳбҹүбһјбһҢбҹӮбһӣбһ…бҹ’бһҡбһҫбһ“бһҠбҹӮбһӣбһўбһ¶бһ…бһҠбҹ„бҹҮбһҹбҹ’бһҡбһ¶бһҷбһ‘бһ·бһ“бҹ’бһ“бһ“бҹҗбһҷбһңбҹҒбһ‘бҹ’бһҡбһҹбһҳбҹ’бһ”бһјбһҡбһ”бҹӮбһ” бһҠбһјбһ…бһҮбһ¶бһўбһҖбҹ’бһҹбһҡ бһҡбһјбһ”бһ—бһ¶бһ– бһ“бһ·бһ„бһҹбһҳбҹ’бһӣбҹҒбһ„бҹ”

MCP бһӮбһ¶бҹҶбһ‘бҹ’бһҡбһҳбһ·бһ“бһҸбҹ’бһҡбһ№бһҳбһҸбҹӮбһҖбһ¶бһҡбһ”бҹүбҹҮбһ–бһ¶бһӣбҹӢбһҮбһ¶бһўбһҖбҹ’бһҹбһҡбһ“бҹ„бҹҮбһ‘бҹҒ бһҸбҹӮбһҗбҹӮбһҳбһ‘бһ¶бҹҶбһ„бһҳбһ¶бһ“бһҹбһҳбһҸбҹ’бһҗбһ—бһ¶бһ–бһҳбҹүбһјбһӣбһ‘бһёбһҳбҹүбһјбһҢбҹӮбһӣ бһҠбҹӮбһӣбһўбһ“бһ»бһүбҹ’бһүбһ¶бһҸбһІбҹ’бһҷбһҳбҹүбһјбһҢбҹӮбһӣбһўбһ¶бһ…бһҠбҹҶбһҺбһҫбһҡбһҖбһ¶бһҡбһҮбһ¶бһҳбһҪбһҷбһҡбһјбһ”бһ—бһ¶бһ– бһҹбһҳбҹ’бһӣбҹҒбһ„ бһ“бһ·бһ„бһ”бҹ’бһҡбһ—бҹҒбһ‘бһ‘бһ·бһ“бҹ’бһ“бһ“бҹҗбһҷбһ•бҹ’бһҹбҹҒбһ„бһ‘бҹҖбһҸбҹ”

## бһҖбһ¶бһҡбһҺбҹӮбһ“бһ¶бҹҶ

бһ“бҹ…бһҖбҹ’бһ“бһ»бһ„бһҳбҹҒбһҡбҹҖбһ“бһ“бҹҒбҹҮ бһўбҹ’бһ“бһҖбһ“бһ№бһ„бһҡбҹҖбһ“бһ–бһёбһҡбһ”бҹҖбһ”бһ”бһ„бҹ’бһҖбһҫбһҸбһҖбһҳбҹ’бһҳбһңбһ·бһ’бһёбһҳбҹүбһјбһӣбһ‘бһёбһҳбҹүбһјбһҢбҹӮбһӣбҹ”

## бһӮбҹ„бһӣбһ”бҹҶбһҺбһ„бһҹбһ·бһҖбҹ’бһҹбһ¶

бһ“бҹ…бһ…бһ»бһ„бһ”бһүбҹ’бһ…бһ”бҹӢбһҳбҹҒбһҡбҹҖбһ“бһ“бҹҒбҹҮ бһўбҹ’бһ“бһҖбһ“бһ№бһ„бһўбһ¶бһ…бҹ–

- бһҷбһӣбҹӢбһҠбһ№бһ„бһўбҹҶбһ–бһёбһҮбһҳбҹ’бһҡбһҫбһҹбһҳбҹүбһјбһҢбҹӮбһӣбһ…бҹ’бһҡбһҫбһ“
- бһўбһ“бһ»бһңбһҸбҹ’бһҸбһҖбһҳбҹ’бһҳбһңбһ·бһ’бһёбһҳбҹүбһјбһҢбҹӮбһӣбһ…бҹ’бһҡбһҫбһ“бҹ”

## бһңбһҸбҹ’бһҗбһ»бһ’бһ¶бһҸбһ»бһҹбһҳбҹ’бһҡбһ¶бһ”бҹӢбһӮбһ¶бҹҶбһ‘бҹ’бһҡбһҳбҹүбһјбһӣбһ‘бһёбһҳбҹүбһјбһҢбҹӮбһӣ

бһҖбһ¶бһҡбһўбһ“бһ»бһңбһҸбҹ’бһҸ MCP бһҳбҹүбһјбһӣбһ‘бһёбһҳбҹүбһјбһҢбҹӮбһӣбһ—бһ¶бһӮбһ…бҹ’бһҡбһҫбһ“бһҡбһҪбһҳбһҳбһ¶бһ“бҹ–

- **бһўбҹ’бһ“бһҖбһ”бҹҶбһӣбҹӮбһ„бһ–бһёбһҳбҹүбһјбһҢбҹӮбһӣбһҮбһ¶бһҖбҹӢбһӣбһ¶бһҖбҹӢ**бҹ– бһ§бһ”бһҖбһҡбһҺбҹҚбһҠбҹӮбһӣбһ”бһҳбҹ’бһӣбҹӮбһ„бһ”бҹ’бһҡбһ—бҹҒбһ‘бһҳбһ·бһҢбһ¶бһҷбһјбһ“бҹ—бһ‘бҹ…бһ‘бҹ’бһҡбһ„бҹӢбһ‘бҹ’бһҡбһ¶бһҷбһҠбҹӮбһӣбһҳбҹүбһјбһҢбҹӮбһӣбһўбһ¶бһ…бһҠбҹҶбһҺбһҫбһҡбһҖбһ¶бһҡбҹ”
- **бһ§бһ”бһҖбһҡбһҺбҹҚбһҹбһҳбҹ’бһҡбһ¶бһ”бҹӢбһҳбҹүбһјбһҢбҹӮбһӣбһҮбһ¶бһҖбҹӢбһӣбһ¶бһҖбҹӢ**бҹ– бһ§бһ”бһҖбһҡбһҺбҹҚбһ–бһ·бһҹбҹҒбһҹбһҠбҹӮбһӣбһҡбһ…бһ“бһ¶бһЎбһҫбһ„бһҹбһҳбҹ’бһҡбһ¶бһ”бҹӢбһҠбҹ„бҹҮбһҹбҹ’бһҡбһ¶бһҷбһҳбҹүбһјбһҢбҹӮбһӣбһҮбһ¶бһҖбҹӢбһӣбһ¶бһҖбҹӢ (бһңбһ·бһ—бһ¶бһӮбһҡбһјбһ”бһ—бһ¶бһ– бһҖбһ¶бһҡбһҠбҹҶбһҺбһҫбһҡбһҖбһ¶бһҡбһҹбһҳбҹ’бһӣбҹҒбһ„)
- **бһҖбһ¶бһҡбһӮбҹ’бһҡбһ”бҹӢбһӮбҹ’бһҡбһ„бһ”бһҡбһ·бһ”бһ‘бһҜбһҖбһҹбһ¶бһҡбһҡбһҪбһҳ**бҹ– бһ”бҹ’бһҡбһ–бҹҗбһ“бҹ’бһ’бһҠбһҫбһҳбҹ’бһ”бһёбһҡбһҖбҹ’бһҹбһ¶бһ”бһҡбһ·бһ”бһ‘бһҖбҹ’бһ“бһ»бһ„бһҳбҹүбһјбһҢбҹӮбһӣбһ•бҹ’бһҹбҹҒбһ„бҹ—бһӮбҹ’бһ“бһ¶
- **бһҖбһ¶бһҡбһ”бһ„бҹ’бһҖбһҫбһҸбһ…бһҳбҹ’бһӣбһҫбһҷ**бҹ– бһҹбһҳбһҸбҹ’бһҗбһ—бһ¶бһ–бһ”бһ„бҹ’бһҖбһҫбһҸбһ…бһҳбҹ’бһӣбһҫбһҷбһҠбҹӮбһӣбһўбһ¶бһ…бһҡбһҪбһҳбһ”бһүбҹ’бһ…бһјбһӣбһҳбҹүбһјбһҢбҹӮбһӣбһ…бҹ’бһҡбһҫбһ“бҹ”

## бһ§бһ‘бһ¶бһ бһҡбһҺбҹҚбһҳбҹүбһјбһҢбҹӮбһӣбһ…бҹ’бһҡбһҫбһ“бҹ– бһңбһ·бһ—бһ¶бһӮбһҡбһјбһ”бһ—бһ¶бһ–

бһ“бҹ…бһ§бһ‘бһ¶бһ бһҡбһҺбҹҚбһҒбһ¶бһ„бһҖбҹ’бһҡбҹ„бһҳ бһҷбһҫбһ„бһ“бһ№бһ„бһңбһ·бһ—бһ¶бһӮбһҡбһјбһ”бһ—бһ¶бһ–бһҳбһҪбһҷ бһ“бһ·бһ„бһҠбһҖбһ–бҹҗбһҸбҹҢбһҳбһ¶бһ“бһ…бҹҒбһүбһҳбһҖбҹ”

### бһҖбһ¶бһҡбһўбһ“бһ»бһңбһҸбҹ’бһҸ C#

```csharp
using ModelContextProtocol.SDK.Server;
using ModelContextProtocol.SDK.Server.Tools;
using ModelContextProtocol.SDK.Server.Content;
using System.Text.Json;
using System.IO;
using System.Threading.Tasks;
using System.Collections.Generic;

namespace MultiModalMcpExample
{
    // Tool for image analysis
    public class ImageAnalysisTool : ITool
    {
        private readonly IImageAnalysisService _imageService;
        
        public ImageAnalysisTool(IImageAnalysisService imageService)
        {
            _imageService = imageService;
        }
        
        public string Name => "imageAnalysis";
        public string Description => "Analyzes image content and extracts information";
          public ToolDefinition GetDefinition()
        {
            return new ToolDefinition
            {
                Name = Name,
                Description = Description,
                Parameters = new Dictionary<string, ParameterDefinition>
                {
                    ["imageUrl"] = new ParameterDefinition
                    {
                        Type = ParameterType.String,
                        Description = "URL to the image to analyze" 
                    },
                    ["analysisType"] = new ParameterDefinition
                    {
                        Type = ParameterType.String,
                        Description = "Type of analysis to perform",
                        Enum = new[] { "general", "objects", "text", "faces" },
                        Default = "general"
                    }
                },
                Required = new[] { "imageUrl" }
            };
        }
        
        public async Task<ToolResponse> ExecuteAsync(IDictionary<string, object> parameters)
        {
            // Extract parameters
            string imageUrl = parameters["imageUrl"].ToString();
            string analysisType = parameters.ContainsKey("analysisType") 
                ? parameters["analysisType"].ToString() 
                : "general";
              // Download or access the image
            byte[] imageData = await DownloadImageAsync(imageUrl);
            
            // Analyze based on the requested analysis type
            var analysisResult = analysisType switch
            {
                "objects" => await _imageService.DetectObjectsAsync(imageData),                "text" => await _imageService.RecognizeTextAsync(imageData),
                "faces" => await _imageService.DetectFacesAsync(imageData),
                _ => await _imageService.AnalyzeGeneralAsync(imageData) // Default general analysis
            };
            
            // Return structured result as a ToolResponse
            // Format follows the MCP specification for content structure
            var content = new List<ContentItem>
            {
                new ContentItem
                {
                    Type = ContentType.Text,
                    Text = JsonSerializer.Serialize(analysisResult)
                }
            };
            
            return new ToolResponse
            {
                Content = content,
                IsError = false
            };
        }
        
        private async Task<byte[]> DownloadImageAsync(string url)
        {
            using var httpClient = new HttpClient();
            return await httpClient.GetByteArrayAsync(url);
        }
    }
    
    // Multi-modal MCP server with image and text processing
    public class MultiModalMcpServer
    {
        public static async Task Main(string[] args)
        {
            // Create an MCP server
            var server = new McpServer(
                name: "Multi-Modal MCP Server",
                version: "1.0.0"
            );
            
            // Configure server for multi-modal support
            var serverOptions = new McpServerOptions
            {
                MaxRequestSize = 10 * 1024 * 1024, // 10MB for larger payloads like images
                SupportedContentTypes = new[]
                {
                    "image/jpeg",
                    "image/png",
                    "text/plain",
                    "application/json"
                }
            };
            
            // Create image analysis service
            var imageService = new ComputerVisionService();
            
            // Register image analysis tools
            server.AddTool(new ImageAnalysisTool(imageService));
            
            // Register a text-to-image tool
            services.AddMcpTool<TextAnalysisTool>();
            services.AddMcpTool<ImageAnalysisTool>();
            services.AddMcpTool<DocumentGenerationTool>(); // Tool that can generate documents with text and images
        }
    }
}
```

- бһ”бһ„бҹ’бһҖбһҫбһҸ `ImageAnalysisTool` бһҠбҹӮбһӣбһўбһ¶бһ…бһңбһ·бһ—бһ¶бһӮбһҡбһјбһ”бһ—бһ¶бһ–бһҠбҹ„бһҷбһ”бҹ’бһҡбһҫбһҹбҹҒбһңбһ¶бһҖбһҳбҹ’бһҳбһҹбҹ’бһҳбһ¶бһ“бһҸбҹӮбһҳбһҪбһҷ `IImageAnalysisService`бҹ”
- бһҖбҹҶбһҺбһҸбҹӢбһҡбһ…бһ“бһ¶бһҹбһҳбҹ’бһ–бҹҗбһ“бҹ’бһ’бһҳбҹүбһ¶бһҹбҹҠбһёбһ“бһҳбҹҒ MCP бһҠбһҫбһҳбҹ’бһ”бһёбһӮбһ¶бҹҶбһ‘бҹ’бһҡбһҖбһ¶бһҡбһҹбҹ’бһ“бһҫбһҹбһ»бҹҶбһ’бҹҶбҹ— бһ“бһ·бһ„бһӮбһ¶бҹҶбһ‘бҹ’бһҡбһ”бҹ’бһҡбһ—бҹҒбһ‘бһҳбһ¶бһҸбһ·бһҖбһ¶бһҡбһјбһ”бһ—бһ¶бһ–бҹ”
- бһ…бһ»бҹҮбһ”бһүбҹ’бһҮбһёбһ§бһ”бһҖбһҡбһҺбҹҚбһңбһ·бһ—бһ¶бһӮбһҡбһјбһ”бһ—бһ¶бһ–бһҮбһ¶бһҳбһҪбһҷбһҳбҹүбһ¶бһҹбҹҠбһёбһ“бһҳбҹҒбҹ”
- бһўбһ“бһ»бһңбһҸбҹ’бһҸбһңбһ·бһ’бһёбһҹбһ¶бһҹбҹ’бһҡбҹ’бһҸбһ‘бһ¶бһүбһҷбһҖбһҡбһјбһ”бһ—бһ¶бһ–бһ–бһё URL бһ“бһ·бһ„бһңбһ·бһ—бһ¶бһӮбһңбһ¶бһҸбһ¶бһҳбһ”бҹ’бһҡбһ—бҹҒбһ‘бһҠбҹӮбһӣбһ”бһ¶бһ“бһҹбҹ’бһ“бһҫбһҹбһ»бҹҶ (бһңбһҸбҹ’бһҗбһ» бһўбһҖбҹ’бһҹбһҡ бһҳбһ“бһ»бһҹбҹ’бһҹбһӣбһҖбҹ’бһҒбһҺбҹҲ бһ”бһүбҹ’бһ…бһјбһӣбһҮбһ¶бһҠбһҫбһҳ)бҹ”

public class AudioTranscriptionTool implements Tool {
    private final AudioProcessor audioProcessor;
    
    public AudioTranscriptionTool(AudioProcessor audioProcessor) {
        this.audioProcessor = audioProcessor;
    }
    
    @Override
    public String getName() {
        return "audioTranscription";
    }
    
    @Override

        return "Transcribes speech from audio files to text";
    }
    
    @Override

        Map<String, Object> schema = new HashMap<>();
        schema.put("type", "object");
        
        Map<String, Object> properties = new HashMap<>();
        
        Map<String, Object> audioUrl = new HashMap<>();
        audioUrl.put("type", "string");
        audioUrl.put("description", "URL to the audio file to transcribe");
        
        Map<String, Object> audioData = new HashMap<>();
        audioData.put("type", "string");
        audioData.put("description", "Base64-encoded audio data (alternative to URL)");
        
        Map<String, Object> language = new HashMap<>();
        language.put("type", "string");
        language.put("description", "Language code (e.g., 'en-US', 'es-ES')");
        language.put("default", "en-US");
        
        properties.put("audioUrl", audioUrl);
        properties.put("audioData", audioData);
        properties.put("language", language);
        
        schema.put("properties", properties);
        schema.put("required", Arrays.asList("audioUrl"));
        
        return schema;
    }
    
    @Override
    public ToolResponse execute(ToolRequest request) {
        try {
            byte[] audioData;
            String language = request.getParameters().has("language") ? 
                request.getParameters().get("language").asText() : "en-US";
                
            // бһ‘бһ‘бһҪбһӣбһҹбҹҶбһЎбҹҒбһ„бһ–бһё URL бһ¬бһ‘бһ·бһ“бҹ’бһ“бһ“бҹҗбһҷбһ•бҹ’бһ‘бһ¶бһӣбҹӢ

                String audioUrl = request.getParameters().get("audioUrl").asText();
                audioData = downloadAudio(audioUrl);

                String base64Audio = request.getParameters().get("audioData").asText();
                audioData = Base64.getDecoder().decode(base64Audio);
            } else {
                throw new ToolExecutionException("Either audioUrl or audioData must be provided");
            }
            
            // бһҠбҹҶбһҺбһҫбһҡбһҖбһ¶бһҡвҖӢбһҹбҹҶбһЎбҹҒбһ„ бһ“бһ·бһ„вҖӢбһ”бһҖбһ”бҹ’бһҡбҹӮ
            Map<String, Object> transcriptionResult = audioProcessor.transcribe(audioData, language);
            
            // бһҸбҹ’бһҡбһЎбһ”бҹӢбһӣбһ‘бҹ’бһ’бһ•бһӣбһҖбһ¶бһҡбһ”бһҖбһ”бҹ’бһҡбҹӮ
            return new ToolResponse.Builder()

        }

class MultiModalResponseHandler:
    """Handler for creating responses that combine text, images, and other modalities"""
    
    def __init__(self, mcp_client):
        self.client = mcp_client
    
    async def create_multi_modal_response(self, 
                                         text_content: str, 
                                         generate_images: bool = False,
                                         image_prompts: Optional[List[str]] = None) -> Dict[str, Any]:
        """
        Creates a response that may include generated images alongside text
        """
        response = {
            "text": text_content,
            "images": []
        }
        
        # бһ”бһ„бҹ’бһҖбһҫбһҸбһҡбһјбһ”бһ—бһ¶бһ– бһ”бҹ’бһҡбһҹбһ·бһ“бһ”бһҫбһ”бһ¶бһ“бһҹбҹ’бһ“бһҫбһҹбһ»бҹҶ
        if generate_images and image_prompts:
            for prompt in image_prompts:
                image_result = await self.client.execute_tool(
                    "imageGeneration",
                    {
                        "prompt": prompt,
                        "style": "realistic",
                        "width": 512,
                        "height": 512
                    }
                )
                
                response["images"].append({
                    "imageData": image_result.result["imageBase64"],
                    "format": image_result.result["format"],
                    "prompt": prompt
                })
        
        return response
