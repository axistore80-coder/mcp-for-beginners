# កម្មវិធីបង្កើតផែនការសិក្សាជាមួយ Chainlit & Microsoft Learn Docs MCP

## អ្វីដែលត្រូវមានជាមុន

- Python 3.8 ឬខ្ពស់ជាង
- pip (កម្មវិធីគ្រប់គ្រងបណ្ណាល័យ Python)
- ការចូលប្រើអ៊ីនធឺណិតដើម្បីភ្ជាប់ទៅកាន់ម៉ាស៊ីនមេ Microsoft Learn Docs MCP

## ការដំឡើង

1. ចម្លង repository នេះ ឬទាញយកឯកសារគម្រោង។
2. ដំឡើងបណ្ណាល័យដែលត្រូវការ៖

   ```bash
   pip install -r requirements.txt
   ```

## ការប្រើប្រាស់

### សេណារីយ៉ូ 1: សំណួรง่ายៗទៅ Docs MCP
កម្មវិធី command-line ដែលភ្ជាប់ទៅម៉ាស៊ីនមេ Docs MCP ស្នើសុំសំណួរ ហើយបង្ហាញលទ្ធផល។

1. បើក script៖
   ```bash
   python scenario1.py
   ```
2. បញ្ចូលសំណួរឯកសាររបស់អ្នកនៅពេល prompt។

### សេណារីយ៉ូ 2: កម្មវិធីបង្កើតផែនការសិក្សា (កម្មវិធីបណ្តាញ Chainlit)
អន្ដរប្រតិបត្តិការ web ដែលប្រើ Chainlit អនុញ្ញាតឲ្យអ្នកប្រើប្រាស់បង្កើតផែនការសិក្សាផ្ទាល់ខ្លួនជារៀងរាល់សប្តាហ៍សម្រាប់មុខវិជ្ជាបច្ចេកទេសណាមួយ។

1. ចាប់ផ្ដើមកម្មវិធី Chainlit៖
   ```bash
   chainlit run scenario2.py
   ```
2. បើក URL ក្នុងកំឡុងជលដដែលនៅក្នុង terminal របស់អ្នក (ឧ. http://localhost:8000) ក្នុងកម្មវិធីប្រើប្រាស់ browser។
3. នៅក្នុងបង្អួច chat បញ្ចូលប្រធានបទសិក្សារបស់អ្នក និងចំនួនសប្តាហ៍ដែលអ្នកចង់សិក្សា (ឧ. "វិញ្ញាបនបត្រ AI-900, 8 សប្តាហ៍")។
4. កម្មវិធីនឹងឆ្លើយតបជាមួយផែនការសិក្សារបស់អ្នករៀងរាល់សប្តាហ៍ ឈានមកជាមួយតំណភ្ជាប់ទៅឯកសារ Microsoft Learn ដែលពាក់ព័ន្ធ។

**អថេរបរិយាកាសដែលត្រូវការជា៖**

ដើម្បីប្រើសេណារីយ៉ូ 2 (កម្មវិធីបណ្តាញ Chainlit ជាមួយ Azure OpenAI) អ្នកត្រូវកំណត់អថេរបរិយាកាសខាងក្រោមនៅក្នុងឯកសារ `.env` នៅថត `python`៖

```
AZURE_OPENAI_CHAT_DEPLOYMENT_NAME=
AZURE_OPENAI_API_KEY=
AZURE_OPENAI_ENDPOINT=
AZURE_OPENAI_API_VERSION=
```

បំពេញតម្លៃទាំងនេះជាមួយព័ត៌មានធនធាន Azure OpenAI របស់អ្នក មុនពេលដំណើរការកម្មវិធី។

> [!TIP]
> អ្នកអាចចេញផ្សាយម៉ូដែលផ្ទាល់ខ្លួនបានយ៉ាងងាយស្រួលដោយប្រើ [Azure AI Foundry](https://ai.azure.com/)។

### សេណារីយ៉ូ 3: ការពិនិត្យឯកសារនៅក្នុងកំណែស្រាយ VS Code ជាមួយម៉ាស៊ីនមេ MCP

ជំនួសការផ្លាស់ប្តូរតាប browser ដើម្បីស្វែងរកឯកសារ អ្នកអាចយក Microsoft Learn Docs ចូលទៅកាន់ VS Code របស់អ្នកដោយផ្ទាល់ ដោយប្រើម៉ាស៊ីនមេ MCP។ វាអនុញ្ញាតឲ្យអ្នក៖
- ស្វែងរក និងអានឯកសារផ្ទាល់ក្នុង VS Code ដោយមិនចេញពីបរិយាកាសកូដ។
- យោងទៅកាន់ឯកសារ និងបញ្ចូលតំណភ្ជាប់ដោយផ្ទាល់ទៅក្នុង README ឬឯកសារបណ្តុះបណ្តាល។
- ប្រើ GitHub Copilot និង MCP រួមគ្នា ដើម្បីមានចរន្តការងារឯកសារដ៏គ្មានចលាចលដែលមាន AI សមត្ថភាព។

**ឧទាហរណ៍នៃការប្រើប្រាស់៖**
- បន្ថែមតំណភ្ជាប់យោងទៅ README យ៉ាងរហ័សពេលកំពុងសរសេរ ឯកសារ ឬគម្រោង។
- ប្រើ Copilot បង្កើតកូដ ហើយប្រើ MCP ស្វែងរក និងយោងឯកសារពាក់ព័ន្ធភ្លាមៗ។
- រក្សាភាពចាប់អារម្មណ៍ក្នុងកំណែស្រាយ និងបន្ថែមផលិតភាព។

> [!IMPORTANT]
> សូមប្រាកដថាអ្នកមានការកំណត់ [`mcp.json`](../../../../../../09-CaseStudy/docs-mcp/solution/scenario3/mcp.json) ត្រឹមត្រូវនៅក្នុងកំណត់ការងាររបស់អ្នក (ទីតាំង `.vscode/mcp.json`)។

## មូលហេតុដែលប្រើ Chainlit សម្រាប់សេណារីយ៉ូ 2?

Chainlit គឺជាស៊ុមវេចខ្ចប់បើកចំហរថ្មីសម្រាប់បង្កើតកម្មវិធីបណ្តាញដែលមានចរន្តសន្ទនា។ វាធ្វើឲ្យងាយស្រួលក្នុងការបង្កើតចំណុចប្រទាក់អ្នកប្រើដែលផ្អែកលើការជជែកបែបជួរដេកដែលភ្ជាប់ទៅសេវាកម្មក្រោយដូចជា Microsoft Learn Docs MCP server។ គម្រោងនេះប្រើ Chainlit ដើម្បីផ្តល់វិធីសាស្រ្តងាយស្រួល និងអន្ដរាយបាន ក្នុងការបង្កើតផែនការសិក្សាផ្ទាល់ខ្លួនក្នុងពេលវេលាផ្លូវចិត្ដពេញលេញ។ ដោយការបោកប្រាស់ Chainlit អ្នកអាចបង្កើត និងប្រើឧបករណ៍បែបទំនាក់ទំនងដែលបង្កើនផលិតភាព និងការសិក្សាបានយ៉ាងឆាប់រហ័ស។

## វាជាកម្មវិធីអ្វី?

កម្មវិធីនេះអនុញ្ញាតឲ្យអ្នកប្រើបង្កើតផែនការសិក្សាផ្ទាល់ខ្លួនដោយការបញ្ចូលប្រធានបទ និងរយៈពេល។ កម្មវិធីនេះធ្វើការបកស្រាយបញ្ចូលរបស់អ្នក ស្នើសុំទៅម៉ាស៊ីនមេ Microsoft Learn Docs MCP សម្រាប់មាតិកាដែលពាក់ព័ន្ធ ហើយរៀបចំលទ្ធផលជាប្លង់រៀងរាល់សប្តាហ៍។ គំនិតណែនាំរៀងរាល់សប្តាហ៍ត្រូវបានបង្ហាញក្នុងជជែក ដែលធ្វើឲ្យងាយស្រួលតាមដាន និងតាមដានជំហានជារយៈពេល។ ការតភ្ជាប់នេះធានាថាអ្នកទទួលបានធនធានសិក្សាមួយដែលមានភាពទាន់សម័យ និងពាក់ព័ន្ធបំផុត។

## សំណួរឧទាហរណ៍

សូមសាកល្បងសំណួរទាំងនេះនៅក្នុងជជែកដើម្បីមើលថាកម្មវិធីឆ្លើយដូចម្ដេច៖

- `វិញ្ញាបនបត្រ AI-900, 8 សប្តាហ៍`
- `សិក្សាអំពី Azure Functions, 4 សប្តាហ៍`
- `Azure DevOps, 6 សប្តាហ៍`
- `វិស្វកម្មទិន្នន័យលើ Azure, 10 សប្តាហ៍`
- `មូលដ្ឋានសុវត្ថិភាព Microsoft, 5 សប្តាហ៍`
- `Power Platform, 7 សប្តាហ៍`
- `សេវាថ្មី Azure AI, 12 សប្តាហ៍`
- `រចនាសម្ព័ន្ធប្រាក់ពពក, 9 សប្តាហ៍`

ឧទាហរណ៍ទាំងនេះបង្ហាញពីភាពបត់បែននៃកម្មវិធីសម្រាប់គោលដៅនិងរយៈពេលសិក្សាផ្សេងៗ។
